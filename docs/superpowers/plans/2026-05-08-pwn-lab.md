# m190 Pwn Lab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `logansandivar.com/lab/` — an in-browser x86 Linux VM (via v86) hosting three original RE crackmes with server-enforced linear gating, public leaderboard, and per-visitor writeup unlocks.

**Architecture:** Static HTML/JS frontend boots v86 + Alpine VM with an RE toolkit baked in. JS injects crackme binaries into the VM's filesystem via 9p virtio. Backend is a single `api.php` file that owns session lifecycle, gates binary delivery, verifies flags, and writes JSON-on-disk leaderboard entries. Crackme C source lives outside the public repo; binaries are copied in.

**Tech Stack:** v86 (vendored), Alpine Linux VM image, PHP 8 (existing host), JSON file storage with `flock` (matches existing forum pattern), vanilla JS, musl-gcc cross-compile for static x86 ELFs.

**Spec:** `docs/superpowers/specs/2026-05-08-pwn-lab-design.md`

**Testing approach:** PHP backend gets curl-based integration tests in `lab/test/` (the project has no test framework; this matches the existing pragmatism). Crackmes are tested by running their corresponding solver scripts and confirming the resulting flag hash matches `EXPECTED_HASHES`. Frontend is verified by manual end-to-end browser tests per the spec.

  ```python
  #!/usr/bin/env python3
  # Reads flags.txt and emits encoded_easy.h, encoded_medium.h, encoded_hard.h.
  # Each header defines `enc[]` and `enc_len` for use by the corresponding crackme.

  import sys

  def write_array(name, data, out):
      out.write(f"const unsigned char enc[] = {{\n    ")
      for i, b in enumerate(data):
          out.write(f"0x{b:02x},")
          out.write("\n    " if (i + 1) % 12 == 0 else " ")
      out.write("\n};\n")
      out.write(f"const unsigned int enc_len = {len(data)};\n")

  def encode_easy(flag):
      # Single-byte XOR, key 0x42
      return bytes(b ^ 0x42 for b in flag.encode())

  def encode_medium(flag, key=b"m190labs"):
      # transform: enc[i] = ((flag[i] << 3) | (flag[i] >> 5)) ^ key[i%8] - i  (in 8-bit)
      out = bytearray()
      for i, c in enumerate(flag.encode()):
          rot = ((c << 3) | (c >> 5)) & 0xFF
          v = (rot ^ key[i % len(key)]) & 0xFF
          v = (v - i) & 0xFF
          out.append(v)
      return bytes(out)

  def encode_hard(flag):
      # Hard tier embeds the flag bytes directly in a bytecode program.
      # gen_encoded.py just emits the flag bytes; the bytecode itself is
      # hand-authored in crackmes/hard.c (see Task 8).
      return flag.encode()

  if __name__ == "__main__":
      with open("flags.txt") as f:
          flags = [line.strip() for line in f if line.strip()]
      assert len(flags) == 3, "flags.txt must have exactly 3 non-empty lines"
      easy, medium, hard = flags
      with open("encoded_easy.h", "w") as f:
          f.write(f"// auto-generated from flags.txt — do not edit\n")
          write_array("enc", encode_easy(easy), f)
      with open("encoded_medium.h", "w") as f:
          f.write(f"// auto-generated from flags.txt — do not edit\n")
          write_array("enc", encode_medium(medium), f)
      with open("encoded_hard.h", "w") as f:
          f.write(f"// auto-generated from flags.txt — do not edit\n")
          write_array("enc", encode_hard(hard), f)
      print(f"Wrote encoded_easy.h ({len(easy)}B flag), encoded_medium.h ({len(medium)}B), encoded_hard.h ({len(hard)}B)")
  ```

  ```c
  // Easy crackme — single-byte xor obfuscation
  // Build: musl-gcc -static -Os -s easy.c -o crackme-easy
  // Solve path: objdump -d, find xor 0x42 loop, dump .rodata, xor in python.

  #include <stdio.h>
  #include <string.h>
  #include "encoded_easy.h"

  static int verify(const unsigned char *input, size_t len) {
      if (len != enc_len) return 0;
      for (size_t i = 0; i < enc_len; i++) {
          unsigned char expected = (unsigned char)(enc[i] ^ 0x42);
          if (input[i] != expected) return 0;
      }
      return 1;
  }

  int main(void) {
      char buf[128];
      printf("flag: ");
      fflush(stdout);
      if (!fgets(buf, sizeof(buf), stdin)) return 1;
      size_t n = strlen(buf);
      if (n > 0 && buf[n - 1] == '\n') buf[--n] = 0;
      if (verify((unsigned char *)buf, n)) {
          puts("yes");
          return 0;
      }
      puts("no");
      return 1;
  ```

- [ ] **Step 2: Write `crackmes/solutions/solve-easy.py`**

  ```python
  #!/usr/bin/env python3
  # Solver for easy crackme — extracts and xor-decodes the flag.
  # Usage: python3 solve-easy.py ../crackme-easy

  import re
  import subprocess
  import sys

  def main(path):
      # Find the .rodata section and dump it.
      out = subprocess.check_output(["objdump", "-s", "-j", ".rodata", path]).decode()
      # The encoded array is whatever bytes are in .rodata; the simplest robust
      # path is to xor each contiguous run of length 16-40 against 0x42 and check
      # for an m190{...} pattern.
      blob = bytearray()
      for line in out.splitlines():
          m = re.match(r"^\s*[0-9a-f]+\s+([0-9a-f ]+?)\s{2}", line)
  ```

  ```c
  // Medium crackme — ptrace anti-debug + per-byte algorithmic transform
  // Build: musl-gcc -static -Os -s medium.c -o crackme-medium
  // Solve path: notice ptrace, neutralize (patch or LD_PRELOAD), then
  //             reverse the transform per byte.

  #include <stdio.h>
  #include <string.h>
  #include <sys/ptrace.h>
  #include "encoded_medium.h"

  static const unsigned char key[8] = { 'm','1','9','0','l','a','b','s' };

  static int verify(const unsigned char *input, size_t len) {
  ```

  ```python
  #!/usr/bin/env python3
  # Solver for medium crackme — reverse the per-byte transform.
  # Usage: python3 solve-medium.py ../crackme-medium
  #
  # The forward transform per byte i is:
  #     rot = rotl(input[i], 3)
  #     v   = rot ^ key[i%8]
  #     enc[i] = v - i
  # Inverse:
  #     v   = enc[i] + i
  #     rot = v ^ key[i%8]
  #     input[i] = rotr(rot, 3)

  import re
  import subprocess
  import sys

  KEY = b"m190labs"

  def rotr8(x, n):
      return ((x >> n) | (x << (8 - n))) & 0xFF

  def main(path):
      out = subprocess.check_output(["objdump", "-s", "-j", ".rodata", path]).decode()
      blob = bytearray()
      for line in out.splitlines():
          m = re.match(r"^\s*[0-9a-f]+\s+([0-9a-f ]+?)\s{2}", line)
          if m:
              blob.extend(bytes.fromhex(m.group(1).replace(" ", "")))
      # Try every offset; decode; look for m190{...}.
      pat = re.compile(rb"m190\{[A-Za-z0-9_]{8,40}\}")
      for start in range(len(blob)):
          for length in range(20, min(50, len(blob) - start) + 1):
              chunk = blob[start:start + length]
              decoded = bytearray()
              ok = True
              for i, b in enumerate(chunk):
                  v = (b + i) & 0xFF
                  rot = v ^ KEY[i % 8]
                  c = rotr8(rot, 3)
                  if c < 0x20 or c > 0x7e:
                      ok = False
                      break
                  decoded.append(c)
              if ok and pat.fullmatch(bytes(decoded)):
                  print(bytes(decoded).decode())
                  return 0
      print("flag not found", file=sys.stderr)
      return 1

  if __name__ == "__main__":
      sys.exit(main(sys.argv[1] if len(sys.argv) > 1 else "../crackme-medium"))
  ```

## Task 8: Author hard crackme (tiny bytecode VM) + solver

**Files (all in `crackmes/`, gitignored):**
- Create: `crackmes/hard.c`
- Create: `crackmes/solutions/solve-hard.py`

- [ ] **Step 1: Write `crackmes/hard.c`**

  The hard tier embeds the flag bytes directly (via `encoded_hard.h`) AND a hand-authored bytecode program in `program[]`. The interpreter walks `program` operating on the input.

  ```c
  // Hard crackme — tiny bytecode VM
  // Build: musl-gcc -static -Os -s hard.c -o crackme-hard
  // Solve path: identify dispatch loop, recover opcode table, transcribe the
  //             program, solve algebraically (or with z3) for input that
  //             produces the expected final state.

  #include <stdio.h>
  #include <string.h>
  #include <stdint.h>
  #include "encoded_hard.h"

  /*
   * Opcodes (1 byte each, may carry inline operands):
   *   0x01 PUSH_IMM imm8       -- push imm8 onto stack
   *   0x02 LOAD_IN  i8         -- push input[i8] onto stack
   *   0x03 LOAD_FL  i8         -- push enc[i8] (target flag byte)
   *   0x04 XOR                 -- a=pop, b=pop, push b^a
   *   0x05 ADD                 -- push b+a
   *   0x06 SUB                 -- push b-a
   *   0x07 ROL3                -- a=pop, push rotl(a,3)
   *   0x08 CMP                 -- a=pop,b=pop; if a!=b, set bad=1
   *   0x09 HALT                -- stop interpreter
   *
   * Program format: a sequence of (opcode, optional imm). The verifier walks
   * the program once per byte index i (i is implicit from the loop), so the
   * program is "what to do for one input byte." `bad` accumulates across all
   * bytes; if any byte fails CMP, bad=1 and verification fails.
   *
   * For each byte i, the program below computes:
   *     rotl(input[i], 3) ^ 0x5b ^ enc[i]   compared to 0
   *     i.e., rotl(input[i], 3) == 0x5b ^ enc[i]
   * which means input[i] == rotr( 0x5b ^ enc[i], 3 ).
   *
   * `enc[i]` is the *target value*, computed at flag-gen time as
   *     enc[i] = rotl(flag[i], 3) ^ 0x5b
   * so the verify routine just needs to check rotl(input[i],3) ^ 0x5b == enc[i].
   *
   * Then the bytecode for ONE byte is:
   *     LOAD_IN  i           ; stack: [input[i]]
   *     ROL3                 ; stack: [rotl(input[i],3)]
   *     PUSH_IMM 0x5b        ; stack: [rotl, 0x5b]
   *     XOR                  ; stack: [rotl ^ 0x5b]
   *     LOAD_FL  i           ; stack: [rotl^0x5b, enc[i]]
   *     CMP                  ; pops two; sets bad if mismatch
   *     HALT
   */

  // The program is one "frame" worth of bytecode; the verifier loops the frame
  // for each input byte index.
  static const unsigned char program[] = {
      0x02, 0x00,        // LOAD_IN  <byte index>  -- byte index patched per loop
      0x07,              // ROL3
      0x01, 0x5b,        // PUSH_IMM 0x5b
      0x04,              // XOR
      0x03, 0x00,        // LOAD_FL  <byte index>
      0x08,              // CMP
      0x09,              // HALT
  };

  static unsigned char rotl8(unsigned char x, int n) {
      return (unsigned char)((x << n) | (x >> (8 - n)));
  }

  // Per-byte target precomputed at flag-gen time:
  //   target[i] = rotl(flag[i], 3) ^ 0x5b
  // Stored in a separate array `target` placed in .rodata. We need this at
  // build time — emit it in encoded_hard.h alongside `enc[]` (which is the
  // flag bytes themselves; we re-encode here).
  // SIMPLER: compute target from `enc` (= flag bytes) at startup. The verifier
  // is what gets reverse-engineered, not the encoding. That keeps the bytecode
  // honest: it really does compare to a rotated/xored value derived from .rodata.

  static int verify(const unsigned char *input, size_t len) {
      if (len != enc_len) return 0;
      // Build target table from enc (= flag bytes baked at compile time)
      // — this happens once at the start of verify, not in the bytecode loop.
      unsigned char target[64];
      for (size_t i = 0; i < enc_len && i < sizeof(target); i++) {
          target[i] = rotl8(enc[i], 3) ^ 0x5b;
      }
      int bad = 0;
      for (size_t i = 0; i < enc_len; i++) {
          // Mini-VM
          uint8_t stack[16];
          int sp = 0;
          size_t pc = 0;
          int halt = 0;
          while (!halt) {
              uint8_t op = program[pc++];
              switch (op) {
                  case 0x01: { uint8_t v = program[pc++]; stack[sp++] = v; break; }
                  case 0x02: { (void)program[pc++]; stack[sp++] = input[i]; break; }
                  case 0x03: { (void)program[pc++]; stack[sp++] = target[i]; break; }
                  case 0x04: { uint8_t a = stack[--sp]; uint8_t b = stack[--sp]; stack[sp++] = b ^ a; break; }
                  case 0x05: { uint8_t a = stack[--sp]; uint8_t b = stack[--sp]; stack[sp++] = (uint8_t)(b + a); break; }
                  case 0x06: { uint8_t a = stack[--sp]; uint8_t b = stack[--sp]; stack[sp++] = (uint8_t)(b - a); break; }
                  case 0x07: { uint8_t a = stack[--sp]; stack[sp++] = rotl8(a, 3); break; }
                  case 0x08: { uint8_t a = stack[--sp]; uint8_t b = stack[--sp]; if (a != b) bad = 1; break; }
                  case 0x09: { halt = 1; break; }
                  default:   { return 0; }
              }
          }
          if (bad) return 0;
      }
      return 1;
  }

  int main(void) {
      char buf[128];
      printf("flag: ");
      fflush(stdout);
      if (!fgets(buf, sizeof(buf), stdin)) return 1;
      size_t n = strlen(buf);
      if (n > 0 && buf[n - 1] == '\n') buf[--n] = 0;
      if (verify((unsigned char *)buf, n)) {
          puts("yes");
          return 0;
      }
      puts("no");
      return 1;
  }
  ```

- [ ] **Step 2: Write `crackmes/solutions/solve-hard.py`**

  ```python
  #     rotl(input[i], 3) ^ 0x5b ?= target[i]
  # where target[i] = rotl(flag[i], 3) ^ 0x5b.
  # Inverting: input[i] = flag[i]. So solving = recovering flag from .rodata.
  ```

  ```html
  <h4>Hard — tiny bytecode VM</h4>
  <p><strong>The trick:</strong> the binary contains a small interpreter that walks a bytecode <code>program[]</code> array, performing operations on your input. The "verifier" you're trying to reverse isn't the C code that's visible in <code>objdump</code> — it's the bytecode the C code is interpreting. To solve, you need to:</p>
  <ol>
      <li>Find the dispatch loop (a big <code>switch</code> on a single byte).</li>
      <li>Recover the opcode table (what each byte means).</li>
      <li>Find <code>program[]</code> in <code>.rodata</code>.</li>
      <li>Re-implement the VM in Python, then either run it forward or solve algebraically.</li>
  </ol>

  <p><strong>The opcode table:</strong></p>
  <pre>0x01 PUSH_IMM imm8       push imm8 onto stack
  0x02 LOAD_IN  i8         push input[i] (i is the implicit byte loop counter)
  0x03 LOAD_FL  i8         push target[i] (precomputed from flag bytes)
  0x04 XOR                 a=pop, b=pop, push b ^ a
  0x05 ADD                 push (b + a) &amp; 0xff
  0x06 SUB                 push (b - a) &amp; 0xff
  0x07 ROL3                a=pop, push rotl(a, 3)
  0x08 CMP                 a=pop, b=pop; if a != b set bad=1
  0x09 HALT                stop interpreter</pre>

  <p><strong>The program (per byte i):</strong></p>
  <pre>LOAD_IN  i           ; stack: [input[i]]
  ROL3                 ; stack: [rotl(input[i], 3)]
  PUSH_IMM 0x5b        ; stack: [rotl, 0x5b]
  XOR                  ; stack: [rotl ^ 0x5b]
  LOAD_FL  i           ; stack: [rotl^0x5b, target[i]]
  CMP                  ; pops two; bad=1 if mismatch
  HALT</pre>

  <p>Reading off: <code>rotl(input[i], 3) ^ 0x5b == target[i]</code>. So <code>input[i] = rotr(target[i] ^ 0x5b, 3)</code>.</p>

  <p>The <code>target</code> array isn't a literal in <code>.rodata</code> — it's computed in <code>verify()</code> at startup as <code>target[i] = rotl(enc[i], 3) ^ 0x5b</code>, where <code>enc[]</code> contains the flag bytes verbatim. So the actual flag is just sitting in <code>.rodata</code> if you look hard enough. The bytecode is doing meaningful work on your input but the answer is hiding in plain sight in another array.</p>

  <p><strong>Solver (the proper way):</strong></p>
  <pre>OPS = {0x01:'PUSH', 0x02:'LIN', 0x03:'LFL', 0x04:'XOR',
         0x05:'ADD', 0x06:'SUB', 0x07:'ROL3', 0x08:'CMP', 0x09:'HALT'}

  def run(prog, i, inp_byte, target_byte):
      pc, sp, stack, bad = 0, 0, [0]*16, 0
      while True:
          op = prog[pc]; pc += 1
          if op == 0x01: stack[sp] = prog[pc]; pc += 1; sp += 1
          elif op == 0x02: stack[sp] = inp_byte; pc += 1; sp += 1
          elif op == 0x03: stack[sp] = target_byte; pc += 1; sp += 1
          elif op == 0x04: a, b = stack[sp-1], stack[sp-2]; sp -= 1; stack[sp-1] = b ^ a
          elif op == 0x07: stack[sp-1] = ((stack[sp-1] &lt;&lt; 3) | (stack[sp-1] &gt;&gt; 5)) &amp; 0xff
          elif op == 0x08: a, b = stack[sp-1], stack[sp-2]; sp -= 2; bad = bad or (a != b)
          elif op == 0x09: return bad
          # ... fill in rest

  # For each byte i, brute-force input bytes 0x20..0x7e, find one where run() returns 0.
  </pre>

  <p><strong>The cheat path:</strong> notice the flag is in <code>.rodata</code> as plain bytes (in <code>enc[]</code>). <code>strings -a -n 16</code> finds it in seconds. This is a feature of the design, not a bug — if you spot it, you've earned the bypass. The lesson: VM-based obfuscation hides the verification logic, but if data is plaintext nearby, you can win without solving the VM. Real malware authors know this and either (a) encode the comparison values too, or (b) compute them at runtime from external sources.</p>

  <p><strong>Real-world tie-in:</strong> bytecode-VM obfuscation is alive and well in commercial protectors (VMProtect, Themida, Code Virtualizer). This is the toy version. The full version: 100+ opcodes, polymorphic dispatch (different VMs per build), and the verification target is computed by yet another VM. <a href="https://research.checkpoint.com/2018/cyber-attack-trends-2018-mid-year-report/">Real samples in the wild use this routinely.</a></p>
  ```

## Task 24: Adversarial check script — confirm gating cannot be bypassed

**Files:**
- Create: `lab/test/test-adversarial.sh`

- [ ] **Step 1: Write the script**

  ```bash
  #!/usr/bin/env bash
  # Adversarial checks per spec section "Adversarial checks (run before shipping)"
  set -e
  BASE="${BASE:-http://localhost:8000}"

  fail() { echo "FAIL: $1"; exit 1; }

  # 1. Direct fetch of medium binary (bypassing api.php) → should be 403 in production.
  #    Locally with `php -S` this WILL be 200 because PHP's built-in server ignores .htaccess.
  #    We accept the local result and note it. Apache will enforce.
  status=$(curl -s -o /dev/null -w '%{http_code}' "$BASE/lab/binaries/crackme-medium")
  echo "direct fetch crackme-medium → $status (expected 403 in production)"

  # 2. Fetch via api.php with garbage token → 401.
  status=$(curl -s -o /dev/null -w '%{http_code}' "$BASE/lab/api.php?action=fetch_binary&tier=medium&token=GARBAGE")
  [ "$status" = "401" ] || fail "fetch_binary garbage token: expected 401, got $status"

  # 3. Fetch via api.php with valid token but no easy solve → 403.
  init=$(curl -s -X POST "$BASE/lab/api.php?action=init")
  TOKEN=$(echo "$init" | sed -n 's/.*"token":"\([a-f0-9]\{64\}\)".*/\1/p')
  status=$(curl -s -o /dev/null -w '%{http_code}' "$BASE/lab/api.php?action=fetch_binary&tier=medium&token=$TOKEN")
  [ "$status" = "403" ] || fail "fetch_binary unmet prereq: expected 403, got $status"

  # 4. Wrong-flag rate limit. 11 wrong submissions in <1 min → 429 on the 11th.
  for i in $(seq 1 11); do
      status=$(curl -s -o /dev/null -w '%{http_code}' -X POST -d "token=$TOKEN&tier=easy&flag=wrong_$i" "$BASE/lab/api.php?action=submit_flag")
      if [ "$i" = "11" ]; then
          [ "$status" = "429" ] || fail "expected 429 on 11th submit, got $status"
      fi
  done

  # 5. Direct access to data dir → 403 in production. Note local result.
  status=$(curl -s -o /dev/null -w '%{http_code}' "$BASE/lab/data/sessions.json")
  echo "direct fetch data/sessions.json → $status (expected 403 in production)"

  # 6. Direct access to writeups → 403 in production.
  status=$(curl -s -o /dev/null -w '%{http_code}' "$BASE/lab/writeups/medium.html")
  echo "direct fetch writeups/medium.html → $status (expected 403 in production)"

  # 7. Search the JS bundle for flag hashes — there should be none.
  if grep -E '[0-9a-f]{64}' lab/lab.js > /dev/null; then
      fail "lab.js contains hex strings that look like hashes — flag hashes should never be in JS"
  fi

  # 8. Check that crackme source isn't in the repo.
  if [ -e crackmes/easy.c ] && git ls-files --error-unmatch crackmes/easy.c >/dev/null 2>&1; then
      fail "crackmes/easy.c is tracked by git — must be gitignored"
  fi

  echo "PASS: adversarial (production-only checks marked above)"
  ```
